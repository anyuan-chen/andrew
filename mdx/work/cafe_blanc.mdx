int solve(vector<int>& nums, int k) {
    int ans = 0;
    const int MOD = 1e9+7;
    int total = 0;
    for (int i : nums) {
        total =(total + (ans + i))%MOD;
    }
    for (int bit = 0; bit < 31 && k > 0; bit++){
        for (int i = 0; i < nums.size() && k > 0; i++){
            if ((nums[i] & (1 << bit)) == 0){
                k-- ;
                total = (total + (1 << bit))%MOD;
            }
        }
    }
    return total%MOD;
}
